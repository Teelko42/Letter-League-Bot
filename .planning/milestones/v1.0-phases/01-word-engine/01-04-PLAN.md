**src/engine/moves.py** -- Implement the Gordon (1994) algorithm:

```python
def find_all_moves(board: Board, rack: list[str], gaddag: GADDAG,
                   mode: str = 'classic') -> list[Move]:
    """
    Entry point: find ALL valid moves on the board with the given rack.
    Returns moves sorted by score descending (WENG-05).

    Algorithm:
    1. For each direction ('H', 'V'):
       a. Compute cross-checks: board.compute_cross_checks(gaddag, direction)
       b. Find anchor squares: board.find_anchors(direction)
       c. For each anchor, run generate_moves_for_anchor
    2. Deduplicate moves (same word, position, direction)
    3. Sort by score descending
    """

def generate_moves_for_anchor(board, rack, gaddag, anchor_row, anchor_col,
                               cross_checks, direction, mode) -> list[Move]:
    """
    Generate all moves passing through this anchor square.

    1. Compute left_limit from board.compute_left_limit
    2. Check if there are existing tiles to the left/above of anchor:
       - If yes: those tiles form a forced prefix. Traverse them through the
         GADDAG (reversed prefix arcs) to get the starting node, then call
         ExtendRight from the anchor position with the GADDAG node after
         crossing the SEPARATOR.
       - If no: call _left_part with empty partial, root node, and left_limit.
    3. Collect all valid placements
    """

def _left_part(node, partial, rack, limit, anchor_row, anchor_col,
               board, gaddag, cross_checks, direction, mode, results):
    """
    Recursive: build the left part by traversing reversed-prefix arcs in GADDAG.

    At each step:
    - If SEPARATOR exists in current node: cross the separator and call
      _extend_right at the anchor position with the suffix node.
    - If limit > 0, for each letter available (rack tiles + blank tile variations):
      - If letter has an arc in the current GADDAG node:
        - Consume tile from rack, add to partial word
        - Recurse with limit-1
        - Restore rack (backtracking)

    Blank tile handling: when a blank ('_' or '') is in the rack, iterate only
    the letters that have outgoing arcs in the current GADDAG node. This prunes
    from 26 to typically 5-10 attempts per node.
    """

def _extend_right(node, partial, placed, rack, row, col,
                  board, gaddag, cross_checks, direction, mode, results):
    """
    Recursive: extend rightward from current position.

    At (row, col):
    - If out of bounds:
      - If TERMINAL in node and at least one tile placed from rack: record valid move
      - Return
    - If cell is occupied:
      - The existing letter must have an arc in the GADDAG node
      - If yes: follow arc, add letter to partial, recurse to next cell
      - If no: return (dead end)
    - If cell is empty:
      - If TERMINAL in node and at least one tile placed from rack: record valid move
      - For each rack tile (including blank->all valid GADDAG arc letters):
        - If letter is in cross_checks[(row, col)] AND has arc in GADDAG node:
          - Consume from rack, mark as placed, add to partial
          - Recurse to next cell
          - Restore rack (backtracking)
    """

def _build_move(board, word_letters, placed_positions, direction,
                gaddag, mode, rack_size) -> Move | None:
    """
    Construct a Move object from the placed tiles and word letters.
    Determine start position from the leftmost/topmost letter.
    Gather cells forming the main word.
    Gather perpendicular words formed by placed tiles.
    Score the move using score_move.
    Return Move object, or None if invalid (safety check).
    """

def _gather_perpendicular_words(board, placed_positions, direction) -> list[list[Cell]]:
    """
    For each newly placed tile, check if it forms a perpendicular word.
    For direction 'H': check vertical (above/below) at each placed tile's column.
    For direction 'V': check horizontal (left/right) at each placed tile's row.
    Gather the full contiguous perpendicular word passing through the placed position.
    Only return words with length >= 2 (single tile = no perpendicular word).

    IMPORTANT: The placed tile itself must be included in the perpendicular word.
    Create a temporary Cell for the placed tile if it hasn't been committed to the
    board yet (move generation is exploratory, not committed).
    """

def _deduplicate_moves(moves: list[Move]) -> list[Move]:
    """
    Remove duplicate moves (same word, start_row, start_col, direction).
    Keep the first occurrence (they should have identical scores).
    Use a set of (word, start_row, start_col, direction) as dedup keys.
    """
```

Key implementation details:
- **Rack representation:** `list[str]`. '_' or '' represents a blank tile. When consuming a blank during LeftPart/ExtendRight, iterate only letters that have arcs in the current GADDAG node (NOT all 26 blindly). This is the key optimization from research Pattern 4.
- **Backtracking:** Use mutable rack with `pop(index)` and `insert(index, tile)` for efficient backtracking. Do NOT copy the rack on each recursive call.
- **Direction abstraction:** Create helper functions:
  - `_next_pos(row, col, direction)` -- returns (row, col+1) for 'H', (row+1, col) for 'V'
  - `_prev_pos(row, col, direction)` -- returns (row, col-1) for 'H', (row-1, col) for 'V'
- **Existing prefix handling:** If tiles exist to the left/above of the anchor, traverse those letters through the GADDAG's reversed-prefix arcs to find the starting node. Then cross the SEPARATOR and call _extend_right. No _left_part needed in this case.
- **Cross-check enforcement:** Only place a letter if it's in `cross_checks.get((row, col), set())`. If the cell has no cross-check entry, treat as empty set (no valid letters -- but this shouldn't happen for anchor-reachable cells).
- **GADDAG traversal:** `node.get(letter)` returns child node or None. `GADDAG.TERMINAL in node` checks word completion. `GADDAG.SEPARATOR in node` checks for prefix-to-suffix transition.
- **First move special case:** Empty board uses center as synthetic anchor. left_limit allows extending left from center. Both H and V moves generated.
- **Move deduplication is critical:** Adjacent anchors can generate the same word. Deduplicate by (word, start_row, start_col, direction) before sorting.

**src/engine/__init__.py:**

```python
from __future__ import annotations
from pathlib import Path
from typing import Optional

from src.engine.gaddag import GADDAG
from src.engine.board import Board
from src.engine.moves import find_all_moves
from src.engine.models import Move, ScoreBreakdown, Cell, MultiplierType, TileUse
from src.engine.tiles import TILE_VALUES, ALPHABET


class GameEngine:
    """
    Public API for the Letter League word engine.
    Stateful: tracks board across turns.
    Downstream phases (difficulty, Discord) consume this API.
    """

    def __init__(self, wordlist_path: Path, cache_path: Path | None = None,
                 rows: int = Board.DEFAULT_ROWS, cols: int = Board.DEFAULT_COLS,
                 multiplier_layout: dict[tuple[int, int], MultiplierType] | None = None,
                 mode: str = 'classic'):
        """
        Initialize engine with wordlist, board config, and scoring mode.
        Builds/loads GADDAG from wordlist (with optional pickle cache).
        Creates Board with given dimensions and multiplier layout.
        """
        self.gaddag = GADDAG.from_wordlist(wordlist_path, cache_path)
        self.board = Board(rows, cols, multiplier_layout)
        self.mode = mode

    def find_moves(self, rack: list[str]) -> list[Move]:
        """
        Find all valid moves for the given rack on the current board.
        Returns moves sorted by score descending.
        The first element is the optimal move (WENG-05).
        """
        return find_all_moves(self.board, rack, self.gaddag, self.mode)

    def best_move(self, rack: list[str]) -> Move | None:
        """
        Return the highest-scoring move, or None if no moves available.
        Convenience wrapper around find_moves()[0].
        """
        moves = self.find_moves(rack)
        return moves[0] if moves else None

    def play_move(self, move: Move) -> None:
        """
        Apply a move to the board: place all rack tiles from the move
        onto the board. Updates board state for subsequent turns.
        Only places tiles that came from the rack (from_rack=True).
        """
        for tile_use in move.rack_tiles_consumed():
            self.board.place_tile(tile_use.row, tile_use.col,
                                  tile_use.letter, tile_use.is_blank,
                                  self.mode)

    def is_valid_word(self, word: str) -> bool:
        """Check if a word is in the dictionary."""
        return self.gaddag.is_valid_word(word)

    @property
    def is_first_turn(self) -> bool:
        """True if no tiles have been placed on the board yet."""
        return self.board.is_empty()
```
