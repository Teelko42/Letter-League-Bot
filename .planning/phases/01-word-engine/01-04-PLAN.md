def find_all_moves(board: Board, rack: list[str], gaddag: GADDAG,
                   mode: str = 'classic') -> list[Move]:
    """
    Entry point: find ALL valid moves on the board with the given rack.
    Returns moves sorted by score descending (WENG-05).

    Algorithm:
    1. Compute cross-checks for both directions (H and V)
    2. Find anchor squares for both directions
    3. For each anchor, run generate_moves_for_anchor
    4. Deduplicate moves (same word, position, direction)
    5. Sort by score descending
    """

def generate_moves_for_anchor(board: Board, rack: list[str], gaddag: GADDAG,
                               anchor_row: int, anchor_col: int,
                               cross_checks: dict, direction: str,
                               mode: str) -> list[Move]:
    """
    Generate all moves passing through this anchor square.

    1. Compute left_limit from board.compute_left_limit
    2. Check if there are existing tiles to the left/above of anchor
       - If yes: those tiles form a forced prefix; go directly to ExtendRight
         with the GADDAG node after traversing the existing prefix
       - If no: call LeftPart with empty partial word, root node, left_limit
    3. Collect all valid placements
    """

def _left_part(node: dict, partial: list[tuple[str, bool]], rack: list[str],
               limit: int, anchor_row: int, anchor_col: int,
               board: Board, gaddag: GADDAG, cross_checks: dict,
               direction: str, mode: str, results: list) -> None:
    """
    Recursive: build the left part of the word by traversing reversed-prefix
    arcs in the GADDAG.

    At each step:
    - Try crossing the SEPARATOR to switch to ExtendRight
    - If limit > 0, for each letter in rack (or blank -> all 26 letters):
      - If letter has an arc in the current GADDAG node:
        - Consume tile from rack, add to partial
        - Recurse with limit-1
        - Restore rack (backtracking)
    """

def _extend_right(node: dict, partial: list[tuple[str, bool]],
                  placed: list[tuple[int, int, str, bool]],
                  rack: list[str], row: int, col: int,
                  board: Board, gaddag: GADDAG, cross_checks: dict,
                  direction: str, mode: str, results: list) -> None:
    """
    Recursive: extend rightward from current position.

    At (row, col):
    - If cell is occupied: the existing letter must have an arc in GADDAG node.
      Follow it and recurse to next cell. Do not consume from rack.
    - If cell is empty:
      - If TERMINAL in current node and at least one tile placed: record valid move
      - For each letter in rack (or blank -> all 26 letters):
        - If letter is in cross_checks[(row, col)] AND has an arc in GADDAG node:
          - Consume from rack, mark as placed, recurse to next cell
          - Restore rack (backtracking)
    - If cell is out of bounds:
      - If TERMINAL in current node and at least one tile placed: record valid move
    """

def _build_move(board: Board, partial: list, placed: list, direction: str,
                gaddag: GADDAG, mode: str, rack_size: int) -> Move | None:

Key implementation details:
- **Rack representation:** Use a list of strings. '_' or '' represents a blank tile. During LeftPart/ExtendRight, when consuming a blank, iterate all 26 letters and try each that has an arc in the GADDAG node.
- **Backtracking:** Use mutable rack (list) with append/pop for efficiency. Do NOT copy the rack on each recursive call.
- **Direction abstraction:** Use helper functions `next_cell(row, col, direction)` and `prev_cell(row, col, direction)` to abstract H vs V traversal. H: col+1/col-1. V: row+1/row-1.
- **Left part with existing tiles:** If the anchor has existing tiles to its left (H) or above (V), traverse those letters through the GADDAG to get the starting node, then call ExtendRight directly (no LeftPart needed — the left part is already on the board).
- **Cross-check enforcement:** Only place a letter if it's in the cross-check set for that cell. This ensures perpendicular validity without rescanning.
- **GADDAG traversal:** The GADDAG node is a dict. `node.get(letter, None)` returns the child node or None. `GADDAG.TERMINAL in node` checks for word completion.

**src/engine/__init__.py:**

```python
class GameEngine:
    """
    Public API for the Letter League word engine.
    Stateful: tracks board across turns.
    Downstream phases (difficulty, Discord) consume this API.
    """

    def __init__(self, wordlist_path: Path, cache_path: Path | None = None,
                 rows: int = 19, cols: int = 27,
                 multiplier_layout: dict[tuple[int, int], MultiplierType] | None = None,
                 mode: str = 'classic'):
        """
        Initialize engine with wordlist, board config, and scoring mode.
        Builds/loads GADDAG from wordlist (with optional pickle cache).
        Creates Board with given dimensions and multiplier layout.
        """
        self.gaddag = GADDAG.from_wordlist(wordlist_path, cache_path)
        self.board = Board(rows, cols, multiplier_layout)
        self.mode = mode

    def find_moves(self, rack: list[str]) -> list[Move]:
        """
        Find all valid moves for the given rack on the current board.
        Returns moves sorted by score descending.
        The first element is the optimal move (WENG-05).
        """
        return find_all_moves(self.board, rack, self.gaddag, self.mode)

    def best_move(self, rack: list[str]) -> Move | None:
        """
        Return the highest-scoring move, or None if no moves available.
        Convenience wrapper around find_moves()[0].
        """
        moves = self.find_moves(rack)
        return moves[0] if moves else None

    def play_move(self, move: Move) -> None:
        """
        Apply a move to the board: place all rack tiles from the move
        onto the board. Updates board state for subsequent turns.
        """
        for tile_use in move.rack_tiles_consumed():
            self.board.place_tile(tile_use.row, tile_use.col,
                                  tile_use.letter, tile_use.is_blank,
                                  self.mode)

    def is_valid_word(self, word: str) -> bool:
        """Check if a word is in the dictionary."""
        return self.gaddag.is_valid_word(word)

    @property
    def is_first_turn(self) -> bool:
        """True if no tiles have been placed on the board yet."""
        return self.board.is_empty()
```
